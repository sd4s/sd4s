select b.value part_no, 
       a.sample_code, 
       a.method, 
       a.cell, 
       a.calculation_formula, 
       a.result_unit, 
       a.result_value_f, 
       a.result_value_s
from sc_lims_dal_ai.ai_method_cell a
left join sc_lims_dal_ai.ai_sample_part_no_groupkey b
on a.sample_code = b.sample_code
where 
a.method like '%TT520%'
and a.parameter like '%TT520%'
order by 1,2
