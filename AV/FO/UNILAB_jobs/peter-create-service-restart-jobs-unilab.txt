rem SCRIPT om HERSTART VAN UNILAB-JOBS ALS SERVICE TE SCHEDULEN.
rem The .lnk file is a standard windows shortcut to a batch file.
rem All this because:
rem service does not know how to start in a folder,
rem cmd.exe does not know how to start in a folder


sc create Unilab-jobs DisplayName= "Unilab-jobs" binpath= "cmd.exe /C C:\Peter\oracle\jobs\dba-restart-db-job-unilab.lnk" type= own start= auto
pause

rem STARTEN van service
sc start Unilab-jobs
pause

rem VERWIJDER de WINDOWS-service weer:
sc delete Unilab-jobs
pause
